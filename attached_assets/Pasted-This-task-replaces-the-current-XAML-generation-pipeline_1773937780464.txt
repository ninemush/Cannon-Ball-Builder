This task replaces the current XAML 
generation pipeline with a layered 
architecture that makes XML correctness 
provable by construction rather than 
enforced by sanitisers.

The fundamental principle: the LLM 
describes intent in structured JSON. 
Code translates that intent to XAML. 
The LLM never writes XML, VB syntax, 
or any string that gets interpolated 
into XML.

This supersedes and replaces the 
approach in tasks #105 and the 
structured expression work. Do not 
implement #105 separately — this 
task is the complete solution.

Create server/xaml/expression-builder.ts

This module converts ValueIntent to 
correctly bracketed VB expressions. 
It is called by the activity resolver 
for every input value.

export function buildExpression(
  value: ValueIntent
): string {
  switch (value.type) {
    case 'literal-string':
      // Wrap in brackets with VB 
      // string quotes
      // & characters are safe inside 
      // brackets
      return `["${value.value.replace(/"/g, '""')}"]`
    
    case 'literal-number':
      return `[${value.value}]`
    
    case 'literal-bool':
      // VB uses True/False
      return value.value.toLowerCase() 
        === 'true' ? '[True]' : '[False]'
    
    case 'variable':
      // Variable reference — bracket wrap
      return `[${value.value}]`
    
    case 'nothing':
      return '[Nothing]'
    
    case 'asset':
      // Asset names are plain strings 
      // for AssetName attribute
      return value.value
    
    case 'credential':
      return value.value
    
    case 'url':
      if (!value.urlParams || 
          Object.keys(value.urlParams)
            .length === 0) {
        return `["${value.value}"]`
      }
      // Build URL with query params
      // using VB string concatenation
      // All inside brackets — & is 
      // safe as VB concat operator
      const params = Object.entries(
        value.urlParams
      ).map(([k, v], i) => 
        `${i === 0 ? '?' : '&'}${k}=`
      )
      // Variables in params get 
      // concatenated with &
      // This produces:
      // ["https://api.com?" & "lat=" 
      //  & str_Lat & "&lon=" & str_Lon]
      return buildUrlExpression(
        value.value, 
        value.urlParams
      )
    
    case 'expression':
      if (value.left && value.operator 
          && value.right !== undefined) {
        const vbOp = toVbOperator(
          value.operator
        )
        return `[${value.left} ${vbOp} ${value.right}]`
      }
      // Fallback: wrap raw value
      return `[${value.value}]`
  }
}

function toVbOperator(op: string): string {
  const map: Record<string, string> = {
    '==': '=',
    '!=': '<>',
    '>=': '>=',
    '<=': '<=',
    '>': '>',
    '<': '<',
    'and': 'And',
    'or': 'Or',
    'not': 'Not',
    '+': '+',
    '-': '-',
    '&': '&'
  }
  return map[op] || op
}

function buildUrlExpression(
  base: string,
  params: Record<string, string>
): string {
  // Produces a bracketed VB expression
  // that concatenates base URL with 
  // query parameters
  // Variables are referenced by name
  // Literals are quoted strings
  // Result is always bracket-wrapped
  const parts = [`"${base}"`]
  Object.entries(params).forEach(
    ([key, val], i) => {
      const separator = i === 0 ? '?' : '&'
      parts.push(`"${separator}${key}="`)
      // If val looks like a variable 
      // name, use it directly
      // Otherwise treat as literal
      if (/^[a-zA-Z_]\w*$/.test(val)) {
        parts.push(val)
      } else {
        parts.push(`"${val}"`)
      }
    }
  )
  return `[${parts.join(' & ')}]`
}

The key guarantee: buildExpression 
ALWAYS returns a string wrapped in 
[...] for any type that will be used 
as a VB expression in XAML. The XML 
parser never sees raw VB operators.

export type ActivitySpec = {
  templateName: string
  displayName: string
  properties: Record<string, string>
  childElements: Record<string, string>
  errorHandling: ErrorHandlingIntent
  children?: ActivitySpec[]
  elseChildren?: ActivitySpec[]
  bodyChildren?: ActivitySpec[]
}

export function resolveStep(
  step: IntentStep,
  catalogService: CatalogService
): ActivitySpec {
  switch (step.stepType) {
    case 'http-get':
    case 'http-post':
      return resolveHttpCall(
        step, catalogService
      )
    case 'assign':
      return resolveAssign(step)
    case 'condition':
      return resolveCondition(step)
    case 'get-asset':
      return resolveGetAsset(step)
    case 'get-credential':
      return resolveGetCredential(step)
    case 'send-email':
      return resolveSendEmail(step)
    case 'log':
      return resolveLog(step)
    case 'delay':
      return resolveDelay(step)
    case 'invoke-workflow':
      return resolveInvoke(step)
    case 'loop-collection':
      return resolveForEach(step)
    case 'retry':
      return resolveRetryScope(step)
    case 'manual':
      return {
        templateName: 'ui:Comment',
        displayName: step.displayName,
        properties: {
          Text: `MANUAL: ${step.manualNote 
            || step.displayName} — 
            implement this step manually`
        },
        childElements: {},
        errorHandling: 'log-continue'
      }
    default:
      return {
        templateName: 'ui:Comment',
        displayName: step.displayName,
        properties: {
          Text: `REVIEW: Unknown step 
            type '${step.stepType}' — 
            implement manually`
        },
        childElements: {},
        errorHandling: 'log-continue'
      }
  }
}

Each resolver function (resolveHttpCall, 
resolveAssign, etc.) calls 
buildExpression() from the Expression 
Builder for every value. No raw string 
interpolation. Every value goes through 
the expression builder.

export function assembleWorkflow(
  intent: WorkflowIntent,
  catalogService: CatalogService
): string {
  // Build variable declarations from 
  // intent.variables — deterministic, 
  // no LLM involved
  const varDeclarations = intent
    .variables.map(buildVariableDecl)
    .join('\n      ')
  
  // Resolve all steps recursively
  const resolvedSteps = intent.steps
    .map(step => resolveStep(
      step, catalogService
    ))
  
  // Assemble each step — wrap in 
  // TryCatch if errorHandling requires
  const activitiesXml = resolvedSteps
    .map(spec => assembleActivitySpec(
      spec, catalogService
    ))
    .join('\n          ')
  
  // Fill workflow header template
  return WORKFLOW_HEADER_TEMPLATE
    .replace('{{workflowName}}', 
      escapeXmlAttr(intent.name))
    .replace('{{variableDeclarations}}', 
      varDeclarations)
    .replace('{{workflowActivities}}', 
      activitiesXml)
}

function assembleActivitySpec(
  spec: ActivitySpec,
  catalogService: CatalogService
): string {
  // Catalog validation
  if (catalogService.isLoaded() && 
      spec.templateName !== 'ui:Comment') {
    const schema = catalogService
      .getActivitySchema(spec.templateName)
    if (!schema) {
      return `<ui:Comment Text="REVIEW: ` +
        `Unknown activity ` +
        `'${escapeXmlAttr(spec.templateName)}' ` +
        `— implement manually" />`
    }
  }
  
  // Get template from catalog or 
  // template builder
  const template = xamlTemplateBuilder
    .getTemplate(spec.templateName)
  
  // Fill template with resolved values
  // All values already bracketed by 
  // Expression Builder
  let xml = fillTemplate(
    template, spec
  )
  
  // Wrap in TryCatch if needed
  if (spec.errorHandling === 
      'log-rethrow' || 
      spec.errorHandling === 
      'retry-rethrow') {
    xml = wrapInTryCatch(
      xml, spec.displayName,
      spec.errorHandling
    )
  }
  
  return xml
}

The WORKFLOW_HEADER_TEMPLATE is the 
fixed XML header — never generated,

namespace declarations.
